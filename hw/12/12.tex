\begin{exercise}
יהי
$V = \RR^4$
ועבור כל
$a \in \mbb{R}$
תהי
\begin{align*}
    g_a\prs{\pmat{x_1 \\ x_2 \\ x_3 \\ x_4}} &= x_1^2 + a x_1 x_2 + 2 x_1 x_3 + 2 x_2^2 + 2 x_2 x_3 + x_3^2 + 2 x_3 x_4 
\end{align*}
תבנית ריבועית על
$V$.

\begin{enumerate}
    \item 
    לכל
    $a \in \mbb{R}$,
    מיצאו תבנית ריבועית
    $f_a$
    על
    $V$
    עבורה
    $g_a\prs{x} = f_a\prs{x, x}$.
    
    \item מיצאו את כל ערכי
    $a \in \mbb{R}$
    עבורם
    $g_a$
    מוגדרת חיובית לחלוטין.
    
    \item
    נניח כעת כי
    $a = 2$.
    מיצאו בסיס
    $B$
    של
    $V$
    עבורו
    $\brs{g}_B$
    מטריצה מהצורה
    $\pmat{I_{n_+} & 0 & 0 \\ 0 & - I_{n_-} & 0 \\ 0 & 0 & 0_{n_0}}$.
\end{enumerate}
\end{exercise}
